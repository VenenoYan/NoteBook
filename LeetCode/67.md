###67 求两个字符串的和
```C++
string rsort(string s){
    string ret;
    int len=s.size()-1;
    while(s[len]=='0'&&len>=0)
        --len;
    if(len<0)
        ret.push_back('0');
    while(len>=0){
        ret.push_back(s[len]);
        --len;
    }
    return ret;
}
string addBinary(string a, string b) {
    int i1,i2,v1,v2,val=0;
    string ret;
    i1=a.size()-1;
    i2=b.size()-1;
    while(i1>=0||i2>=0){
        v1=i1>=0?a[i1]-48:0;
        v2=i2>=0?b[i2]-48:0;                //char to int 时自动转换，但是ASCII码注意
        if(v1+v2+val==0){
            ret.push_back('0');
            val = 0;
        }
        else if(v1+v2+val==1){
            ret.push_back('1');
            val = 0;
        }else if(v1+v2+val==2){
            ret.push_back('0');
            val = 1;
        }else{
            ret.push_back('1');
            val = 1;
        }
        if(i1>=0)
            --i1;
        if(i2>=0)
            --i2;
    }
    if(val==1)
        ret.push_back('1');
    return rsort(ret);
}
```
